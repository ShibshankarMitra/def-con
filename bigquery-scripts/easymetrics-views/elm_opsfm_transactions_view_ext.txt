SELECT
    PARTITION_DATE,
    EVENT_ID,
    USER_ID,
    EVENT_TYPE,
    START_LOCATION,
    START_LOCATION_TYPE_CODE,
    END_LOCATION,
    END_LOCATION_TYPE_CODE,
    DC_NUMBER,
    EVENT_TIMESTAMP,
    LPN_NUMBER,
    QUANTITY,
    ORDER_TYPE,
    SKU,
    UOM,
    TRACE_ID
  FROM
    `elm_integration.opsfm_transactions`
  WHERE
    PARTITION_DATE BETWEEN DATE_SUB(CURRENT_DATE(), INTERVAL 14 DAY)
    AND CURRENT_DATE()
  ORDER BY
    EVENT_TIMESTAMP DESC
