SELECT
  *
FROM (
  select USER_ID, USER_NAME, PUNCH_DATE, PUNCH_TYPE, PUNCH_REASON, ORDER_NUMBER, PUNCH_LOCAL_TIME, PUNCH_UTC_TIME, DC_NUMBER
  FROM `elm_integration.opsfm_ta_punches`
  where PUNCH_DATE BETWEEN DATE_SUB(CURRENT_DATE(), INTERVAL 14 DAY) AND CURRENT_DATE()
  UNION ALL
  select USER_ID, USER_NAME, PUNCH_DATE, PUNCH_TYPE, null as PUNCH_REASON, null as ORDER_NUMBER, PUNCH_LOCAL_TIME, PUNCH_UTC_TIME, DC_NUMBER
  FROM `elm_integration.full_elm_3pl_cico_punches`
  where PUNCH_DATE BETWEEN DATE_SUB(CURRENT_DATE(), INTERVAL 14 DAY) AND CURRENT_DATE()
)
ORDER BY
  PUNCH_DATE,
  USER_ID,
  PUNCH_LOCAL_TIME,
  ORDER_NUMBER
